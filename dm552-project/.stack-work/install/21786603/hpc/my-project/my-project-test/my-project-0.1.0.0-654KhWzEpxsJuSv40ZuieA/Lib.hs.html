   56 --used for simulateGame
   57 dropSuperPrefix :: String -> String
   58 dropSuperPrefix card | isPrefixOf "Super " card = drop (length "Super ") card
   59 dropSuperPrefix card = card

   61 --used for simulateGame
   62 whoseTurn :: State -> [Move]
   63 whoseTurn ([c1,c2,c3,c4,_], pawns, _, 0, usedSuper, _) = possibleMoves player1State (0)
   64   where player1State = ([c1,c2], pawns, usedSuper)
   65 whoseTurn ([c1,c2,c3,c4,_], _, pawns, 1, _, usedSuper) = possibleMoves player2State (1)
   66   where player2State = ([c3,c4], pawns, usedSuper)

   68   --used for simulateGame
   69 insideBoard :: Position -> Bool
   70 insideBoard (x, y) = x >= 0 && x <= 4 && y >= 0 && y <= 4
   71 
   72 --used for simulateGame
   73 checkWin :: ([Position],[Position]) -> ([Position],[Position])
   74 checkWin (pawns1, pawns2)
   75   | length pawns1 == 1 && elem (4,2) pawns1 = (pawns1,[])
   76   | length pawns2 == 1 && elem (0,2) pawns2 = ([],pawns2)
   77   | otherwise = (pawns1,pawns2)

   81 removeItem list item = delete item list

   83 -- used for generateGame
   84 nextMove :: State -> StdGen -> Integer -> [Move] -> [Move]
   85 nextMove state gen movesNeed finalList
   86   | ((length finalList == fromIntegral movesNeed) || null pawns1 || null pawns2) = finalList
   87   | otherwise = nextMove upState upGen movesNeed upList
   88   where upState = applyMove state (last upList)
   89         upList = finalList ++ upMoves
   90         (upMoves, upGen) = generateSomething (whoseTurn state) [] gen 1
   91         (cards, pawns1, pawns2, turn, super1, super2) = state

   93 -- used for generateGame
   94 generateSomething :: (Eq a, RandomGen t) => [a] -> [a] -> t -> Integer -> ([a],t)
   95 generateSomething fromList toList genNum n
   96   | length toList == fromIntegral n = (toList,genNum)
   97   | otherwise = generateSomething (removeItem fromList (fromList !! i)) ((fromList !! i) : toList) g n
   98   where
   99     l = length fromList -1
  100     (i,g) = randomR (0,l) genNum

  102 -- used to split playCards moves for player1 and player2
  103 splitList :: [a] -> ([a], [a])
  104 splitList myList = splitAt ((length myList + 1) `div` 2) myList

  106 -- makes list of all states aka possible games
  107 possibilities :: State -> Integer -> [State] -> [State]
  108 possibilities state nMoves stateList
  109   | nMoves == 0 = stateList ++ [state]
  110   | null pawns1 || null pawns2 = stateList ++ [state]
  111   | otherwise = do
  112     let moves = whoseTurn state
  113     let upState = map (applyMove state) moves
  114     concatMap(\upd -> possibilities upd (nMoves-1) stateList) upState
  115     where (cards, pawns1, pawns2, turn, super1, super2) = state

  119 -- determines moves based on used supers and player turn
  120 possibleMoves :: PlayerState -> Integer -> [Move]
  121 possibleMoves (cards, pawns, True) 0 = normalMoves
  122   where normalMoves = concatMap (cardMoves pawns) cards
  123 possibleMoves (cards, pawns, False) 0 = normalMoves ++ superMoves pawns cards
  124   where normalMoves = concatMap (cardMoves pawns) cards
  125 possibleMoves (cards, pawns, True) 1 = normalMoves
  126   where normalMoves = concatMap (cardMovesP2 pawns) cards
  127 possibleMoves (cards, pawns, False) 1 = normalMoves ++ superMoves pawns cards
  128   where normalMoves = concatMap (cardMovesP2 pawns) cards

  130 -- deals with super moves
  131 superMoves :: [Position] -> [Card] -> [Move]
  132 superMoves pawns cards =
  133   [(pos1, pos2, "Super " ++ card) |
  134    pos1 <- pawns,
  135    pos2 <- possibleSuperMoves pos1,
  136    insideBoard pos2,
  137    card <- cards]
  138 
  139 -- finds all possible card moves for player1 and determines if they are legal
  140 cardMoves :: [Position] -> Card -> [Move]
  141 cardMoves pawns card = legalMoves
  142   where moves = concatMap (pawnMoves card) pawns
  143         legalMoves = filter (\(_, pos2, _) -> notElem pos2 pawns) moves
  144 
  145 -- finds all possible card moves for player2 and determines if they are legal
  146 cardMovesP2 :: [Position] -> Card -> [Move]
  147 cardMovesP2 pawns card = legalMoves
  148   where moves = concatMap (pawnMovesP2 card) pawns
  149         legalMoves = filter (\(_, pos2, _) -> notElem pos2 pawns) moves
  150 
  151 -- all possible pawn moves for player1; checks if they are inside the board
  152 pawnMoves :: Card -> Position -> [Move]
  153 pawnMoves card pos1 =
  154   [(pos1, pos2, card) | pos2 <- halfA, insideBoard pos2]
  155   where (halfA,halfB) = splitList (playCards pos1 card)
  156 
  157  -- all possible pawn moves for player2; checks if they are inside the board
  158 pawnMovesP2 :: Card -> Position -> [Move]
  159 pawnMovesP2 card pos1 =
  160     [(pos1, pos2, card) | pos2 <- halfB, insideBoard pos2]
  161     where (halfA,halfB) = splitList (playCards pos1 card)

  163 -- applies given move to the current state and updates the state
  164 applyMove :: State -> Move -> State
  165 applyMove state move = newState
  166   where (cards, pawns1, pawns2, turn, super1, super2) = state
  167         (pos1, pos2, card) = move
  168         newCards = applyCardMove turn cards (dropSuperPrefix card)
  169         (newPawns1, newPawns2) = applyPawnMove turn pawns1 pawns2 pos1 pos2
  170         (finalPawns1, finalPawns2) = checkWin (newPawns1, newPawns2)
  171         (newUsedSuper1, newUsedSuper2) = applySuperMove turn super1 super2 card
  172         newTurn = mod (turn+1) 2
  173         newState = (newCards, finalPawns1, finalPawns2, newTurn, newUsedSuper1, newUsedSuper2)
  174 
  175 -- applies card moves and rearranges the cards appropriately
  176 applyCardMove :: Integer -> [Card] -> Card -> [Card]
  177 applyCardMove 0 [c1,c2,c3,c4,c5] card
  178   | card == c1 = sort [c2,c5] ++ [c3,c4,card]
  179   | card == c2 = sort [c1,c5] ++ [c3,c4,card]
  180 applyCardMove 1 [c1,c2,c3,c4,c5] card
  181   | card == c3 = [c1,c2] ++ sort [c4,c5] ++ [card]
  182   | card == c4 = [c1,c2] ++ sort [c3,c5] ++ [card]
  183 
  184 -- applies supermoves and changes super status appropriately
  185 applySuperMove :: (Eq a, Num a) => a -> Bool -> Bool -> String -> (Bool, Bool)
  186 applySuperMove turn super1 super2 card
  187   | "Super " `isPrefixOf` card =
  188     case (turn, super1, super2) of
  189       (0, False, _) -> (True, super2)
  190       (1, _, False) -> (super1, True)
  191 applySuperMove _ super1 super2 _ = (super1, super2)
  192 
  193 applyPawnMove :: Integer -> [Position] -> [Position] -> Position -> Position -> ([Position], [Position])
  194 applyPawnMove 0 pawns1 pawns2 pos1 pos2 = (playerPawns, opponentPawns)
  195   where (playerPawns, opponentPawns) = applyPawnMove1 pawns1 pawns2 temple0 pos1 pos2
  196         temple0 = (2, 4)
  197 applyPawnMove 1 pawns1 pawns2 pos1 pos2 = (opponentPawns, playerPawns)
  198   where (playerPawns, opponentPawns) = applyPawnMove1 pawns2 pawns1 temple1 pos1 pos2
  199         temple1 = (2, 0)
  200 
  201 applyPawnMove1 (playerMaster:playerPawns) (opponentMaster:opponentPawns) temple pos1 pos2
  202   | playerMaster == temple = (temple:playerPawns, [])
  203   | pos2 == opponentMaster = (movePawn (playerMaster:playerPawns) pos1 pos2, [])
  204   | otherwise = (movePawn (playerMaster:playerPawns) pos1 pos2, opponentMaster:(filter (/= pos2) opponentPawns))
  205 
  206 -- adjusts pawns after move
  207 movePawn :: [Position] -> Position -> Position -> [Position]
  208 movePawn (master:pawns) pos1 pos2
  209   | master == pos1 = pos2:pawns
  210   | otherwise = master:(sort $ map replacePawn pawns)
  211       where replacePawn = \p -> if p == pos1 then pos2 else p

  220 convertState ::  Maybe ([String], [(Integer,Integer)], [(Integer, Integer)], Integer, Bool, Bool) -> State
  221 convertState x = (cards, pawns1, pawns2, turn, super1, super2)

  226 convertMove x = (from, to, card)

  229 invalidState :: State -> Bool
  230 invalidState (cards, pawns1, pawns2, turn, super1, super2)
  231   | length cards /= 5 = True
  232   | null pawns1 || length pawns1 > 5 = True
  233   | null pawns2 || length pawns2 > 5 = True
  234   | not (null (intersect pawns1 pawns2)) = True
  235   | (any ((==False) . (`elem` cardsList))) cards = True
  236   | (any ((==False) . insideBoard)) pawns1 = True
  237   | otherwise = False

  239 -- checks for instances of an invalidformat
  240 invalidFormat :: State -> Bool
  241 invalidFormat state
  242   | cardsNotSorted state = True
  243   | pawnsNotSorted state = True
  244   | otherwise = False
  245 
  246 -- checks if cards are sorted appropriately
  247 cardsNotSorted :: State -> Bool
  248 cardsNotSorted (cards, pawns1, pawns2, turn, super1, super2) =
  249   cards /= sortedCards
  250   where [c1,c2,c3,c4,c5] = cards
  251         sortedCards = sort[c1,c2] ++ sort[c3,c4] ++ [c5]
  252 
  253 -- sorts cards according to assignment requirements
  254 sortCards :: [Card] -> [Card]
  255 sortCards [c1,c2,c3,c4,c5] = sort[c1,c2] ++ sort[c3,c4] ++ [c5]
  256 
  257 -- checks if pawns are sorted appropriately
  258 pawnsNotSorted :: State -> Bool
  259 pawnsNotSorted (cards, pawns1, pawns2, turn, super1, super2)
  260   | pawns1 /= head pawns1 : (sort . tail) pawns1 = True
  261   | pawns2 /= head pawns2 : (sort . tail) pawns2 = True
  262   | otherwise = False

  266 simulateGameAux :: State -> [Move] -> String
  267 simulateGameAux state [] = show state
  268 simulateGameAux state (m:ms) =
  269   case whoseTurn state of
  270     moves | elem m moves -> simulateGameAux (applyMove state m) ms
  271     _ -> "InvalidMove" ++ " " ++ show m

  274 countGamesAux :: State -> Integer -> String
  275 countGamesAux state nMoves = show (length games, p1wins, p2wins)
  276   where games = possibilities state nMoves [] --
  277         p1wins = length $ filter (\(_,_,pawns2,_,_,_)-> null pawns2) games -- filters player1 wins from list of states
  278         p2wins =  length $ filter (\(_,pawns1,_,_,_,_)-> null pawns1) games -- filters player2 wins from lists of states

  280 simulateGame :: FilePath -> IO String
  281 simulateGame path = do
  282   file <- readFile path
  283   if file == "" then return "InvalidFormat"
  284     else do
  285     let stringArray = lines file
  286     let fileState = readMaybe (head stringArray) :: Maybe ([String], [(Integer,Integer)], [(Integer, Integer)], Integer, Bool, Bool)
  287     let realState = convertState fileState
  288     if isNothing fileState then return "InvalidFormat"
  289     else do
  290       if invalidState realState then return "InvalidState"
  291       else do
  292         if invalidFormat realState then return "InvalidFormat"
  293         else do
  294           let listofMaybeMoves = map maybeMove (tail stringArray)
  295           if any isNothing listofMaybeMoves then return (show realState)
  296           else do
  297             let listOfMoves = map convertMove listofMaybeMoves
  298             return (simulateGameAux realState listOfMoves)

  300 generateGame :: Integer -> Integer -> String
  301 generateGame s n = do
  302   --random cards, sorted
  303   let rando = mkStdGen (fromIntegral s)
  304   let (cards, genNum) = generateSomething cardsList [] rando 5
  305   let sortedCards = sortCards cards
  306 
  307   -- default pawn positions and supers, turn is random
  308   let pawns1 = [(0,2),(0,0),(0,1),(0,3),(0,4)]
  309   let pawns2 = [(4,2),(4,0),(4,1),(4,3),(4,4)]
  310   let (turn, genNum2) = randoNo 1 genNum :: (Integer, StdGen)
  311   let super1 = False
  312   let super2 = False
  313   let genState = (sortedCards, pawns1, pawns2, turn, super1, super2)
  314 
  315   -- converts state to pass in nextMove and whoseTurn
  316   let stringState = lines (show genState)
  317   let realState = convertState (readMaybe (head stringState))
  318 
  319   -- generates moves list based on the generated state
  320   let genMoves = nextMove realState genNum2 n []
  321 
  322   -- game output; lists inital state and possible moves for that inital state
  323   let genGame = show genState : map show genMoves
  324 
  325   unlines genGame

  327 countGames :: Integer -> FilePath -> IO String
  328 countGames n path = do
  329   file <- readFile path
  330   if null file then return "InvalidFormat"
  331   else do
  332     let stringArray = lines file
  333     let fileState = readMaybe (head stringArray) :: Maybe ([String], [(Integer,Integer)], [(Integer, Integer)], Integer, Bool, Bool)
  334     let realState = convertState fileState
  335     if isNothing fileState then return "InvalidFormat"
  336     else do
  337       if invalidState realState then return "InvalidState"
  338       else do
  339         if invalidFormat realState then return "InvalidFormat"
  340         else do
  341           return $ countGamesAux realState n
